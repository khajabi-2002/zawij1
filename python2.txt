def is_leap(year):
    
    # variable to check the leap year
    leap = False
    
    # using nested if statements
    if (year % 4) == 0:
        if (year % 100) == 0:
            if (year % 400) == 0:
                leap = True
            else:
                leap = False
        else:
            leap = True
    else:
        leap = False
    
    return leap
